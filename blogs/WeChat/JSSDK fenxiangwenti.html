JSSDK 分享问题

MuYan2020-12-11WeChat微信公众号微信

JSSDK

updateAppMessageShareData 等分享 api

分享接口 link 参数, 无法进行以下参数携带的方式，此分享方式微信开发者工具 IDE 会提示分享成功，真机却是失败或无提示

link: '分享的地址链接?url='index?id=1''

解决办法：url 数据处理，如 url 参数进行 base64 加密处理\转义处理，并增加解密\转义逻辑

  • 转义处理
decodeURIComponent() 函数可对 encodeURIComponent() 函数编码的 URI 进行解码。
  • base64 加密

js-base64 地址open in new window

也可自行下载 js-base64 到本地

    return cc < 0x80
      ? c
      : cc < 0x800
      ? _fromCC(0xc0 | (cc >>> 6)) + _fromCC(0x80 | (cc & 0x3f))

  switch (cccc.length) {
    case 4:
      var cp =
          ((0x07 & cccc.charCodeAt(0)) << 18) |
          ((0x3f & cccc.charCodeAt(1)) << 12) |
          ((0x3f & cccc.charCodeAt(2)) << 6) |
          (0x3f & cccc.charCodeAt(3)),

const atobPolyfill = (asc) => {
  // console.log('polyfilled');
  asc = asc.replace(/\s+/g, "");
  if (!b64re.test(asc)) throw new TypeError("malformed base64.");
  asc += "==".slice(2 - (asc.length & 3));
  let u24,
    bin = "",
    r1,
    r2;
  for (let i = 0; i < asc.length; ) {
    u24 =
      (b64tab[asc.charAt(i++)] << 18) |
      (b64tab[asc.charAt(i++)] << 12) |
      ((r1 = b64tab[asc.charAt(i++)]) << 6) |
      (r2 = b64tab[asc.charAt(i++)]);
    bin +=
      r1 === 64
        ? _fromCC((u24 >> 16) & 255)
        : r2 === 64
        ? _fromCC((u24 >> 16) & 255, (u24 >> 8) & 255)
        : _fromCC((u24 >> 16) & 255, (u24 >> 8) & 255, u24 & 255);
  }
  return bin;
};
